ENV USER="alpine"
ENV HOME="/home/${USER}"
ENV PASSWORD="alpine"

# add sudo user
RUN apk add --no-cache \
  bash \
  sudo && \
  addgroup -S sudo && \
  sed -i -r 's/# %sudo/%sudo/' /etc/sudoers && \
  addgroup -g 1000 -S ${USER} && \
  adduser -h ${HOME} -s /bin/bash -G ${USER} -S -D -u 1000 ${USER} && \
  addgroup ${USER} sudo && \
  echo "${USER}:${PASSWORD}" | chpasswd && \
  chown -R ${USER}:${USER} ${HOME}

COPY --chown=1000 .bashrc ${HOME}
USER ${USER}
WORKDIR ${HOME}
